clear variables;
num = [1 0];
den = [1 -0.9];

%raspuns la impuls
h = impz(num, den);

figure,
subplot(111), stem(0:length(h)-1, h), grid, title('Impulse response');

%caracteristicile raspunsului la freceventa
figure,
freqz(num, den, 2^10);

%stabilitatea filtrului
%metoda 1
%RASPUNS: pol in interiorul cercului unitate => filtrul este stabil
figure,
zplane(num, den);

%metoda 2
if isstable(num, den)
disp('Filtrul este stabil')
else
disp('Filtrul este instabil')
end

%metoda 3
poles = roots(den);
stabil = 0;
for i=1:length(poles)
	if abs(poles(i))>1
		stabil = 1;
		break
	end
end

if stabil == 0
	disp('Filtrul este stabil')
else
	disp('Filtrul este instabil')
end

%iesirea filtrului la excitatie
n = 0:100;
x = sin(2*pi*0.05*n);
y = filter(num, den, x);

figure,
subplot(211), stem(n, x);
subplot(212), stem(0:length(y)-1, y)
