clear variables;
n = 0:20;
x1 = 1.5 * cos(2*pi*0.1*n+pi/4);
X1 = length(x1);
x2 = abs(10-n);
X2 = length(x2);
y = conv(x1, x2);
N = X1 + X2 - 1;
disp(['Lungimea convolutiei liniare este: ' num2str(N)])

figure,
subplot(221), stem(0:X1-1, x1), grid, xlabel('n'), ylabel('Amplitudine'), title('x1(n)'),
subplot(222), stem(0:X2-1, x2), grid, xlabel('n'), title('x2(n)'),
subplot(212), stem(0:N-1, y), grid, xlabel('n'), ylabel('Amplitudine'), title('y(n)=x1(n) \otimes x2(n)')
